Исправленный код:
FILE *f = fopen(name, “w”);
if (f == NULL) perror ("Error opening file");//Ошибка при открытии файла 
else
{
    try{
        // Тут код, при котором может быть выброшено исключение
    } catch (...) {//Отлов любого исключения
        fclose (f);//Закрытие файла
        throw;//Передать исключение дальше
    }
    fclose (f);
}
